\begin{problem} Which of the following ODEs is linear?
\begin{multipleChoice}
\choice{$y'' - y^2 = 0$}
\choice{$\sin y = y'$}
\choice[correct]{$y''+t y = 0$}
\choice{$\sqrt{1+y^2}=y'$}
\end{multipleChoice}
\end{problem}

\begin{problem}
   If $y'' - y = 0$, then $y = \answer{c_1 \cosh(t) + c_2 \sinh(t)}$ where $c_1$ and $c_2$ are \answer{constants}.
\end{problem}
