\begin{problem} Which of the following ODEs is linear?
\begin{multipleChoice}
\choice{$y'' - y^2 = 0$}
\choice{$\sin y = y'$}
\choice[correct]{$y''+t y = 0$}
\choice{$\sqrt{1+y^2}=y'$}
\end{multipleChoice}
\end{problem}
